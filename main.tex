\subsection{Rules}

\subsubsection{Sum}

\[ \frac{d}{dx}(f + g)(x) = \frac{d}{dx}f(x) + \frac{d}{dx}g(x) \]

\subsubsection{Product}

\[ \frac{d}{dx}(fg)(x) = g(x)\frac{d}{dx}f(x) + f(x)\frac{d}{dx}g(x) \]

\subsubsection{Quotient}

\[ \frac{d}{dx}\left(\frac{f}{g}\right)(x) = \frac{g(x)\frac{d}{dx}f(x) - f(x)\frac{d}{dx}g(x)}{g^2(x)} \]

\subsubsection{Chain}

\[ \frac{d}{dx}f\left(g(x)\right) = \frac{d}{dx}f\left(g(x)\right)\frac{d}{dx}g(x) \]
